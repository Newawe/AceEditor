      function setCookie(cname, cvalue, exdays) {
        var d = new Date();
        d.setTime(d.getTime() + (exdays*24*60*60*1000));
        var expires = "expires="+d.toUTCString();
        document.cookie = cname + "=" + cvalue + "; " + expires;
      }

      function download() {
        var code = editor.getValue();
        var encodedData = btoa(code);
        //Language specific switch. HTML downloads "download.html", PHP downloads "download.php", etc. Defaults to .txt
        switch(lang.value){
          case "html":
            downloadURI("data:text/html;base64,"+encodedData, "download.html");
            break;
          case "php":
            downloadURI("data:text/html;base64,"+encodedData, "download.php");
            break;
          case "python":
            downloadURI("data:text/html;base64,"+encodedData, "download.py");
            break;
          case "javascript":
            downloadURI("data:text/html;base64,"+encodedData, "download.js");
            break;
          case "css":
            downloadURI("data:text/html;base64,"+encodedData, "download.css");
            break;
          case "mysql":
            downloadURI("data:text/html;base64,"+encodedData, "download.sql");
            break;
          case "java":
            downloadURI("data:text/html;base64,"+encodedData, "download.java");
            break;
          case "perl":
            downloadURI("data:text/html;base64,"+encodedData, "download.pl");
            break;
          default:
            downloadURI("data:text/html;base64,"+encodedData, "download.txt");
            break;
        }
        //data:application/octet-stream;charset=utf-16le;base64,
      }
      function downloadURI(uri, name) {
        var link = document.createElement("a");
        link.download = name;
        link.href = uri;
        link.click();
      }
